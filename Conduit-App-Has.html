newtype Name = Name Text deriving (Show)
newtype NameBox = NameBox { nm :: Name }

instance Has' Name NameBox where 
 obtain :: NameBox -> Name
 obtain = nm

test :: (Has Name c m, MonadIO m) -> m ()
test = do
  name <- grab @Name
  print name

main :: IO ()
main = runReaderT test (NameBox $ Name "hi")
